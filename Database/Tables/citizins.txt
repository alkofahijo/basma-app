CREATE TABLE `citizens` (
  `id` int unsigned NOT NULL AUTO_INCREMENT,
  `name_ar` varchar(150) COLLATE utf8mb4_bin NOT NULL,
  `name_en` varchar(150) COLLATE utf8mb4_bin NOT NULL,
  `mobile_number` varchar(20) COLLATE utf8mb4_bin NOT NULL,
  `government_id` int unsigned NOT NULL,
  `reports_completed_count` int NOT NULL DEFAULT '0',
  `is_active` tinyint(1) NOT NULL DEFAULT '1',
  `created_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uq_citizens_mobile` (`mobile_number`),
  KEY `idx_citizens_government` (`government_id`),
  CONSTRAINT `fk_citizens_government` FOREIGN KEY (`government_id`) REFERENCES `governments` (`id`) ON DELETE RESTRICT ON UPDATE RESTRICT
) ENGINE=InnoDB AUTO_INCREMENT=5 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_bin;
