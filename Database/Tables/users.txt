USE `basmadb`;

CREATE TABLE `users` (
  `id` int unsigned NOT NULL AUTO_INCREMENT,
  `username` varchar(150) COLLATE utf8mb4_bin NOT NULL,
  `hashed_password` varchar(255) COLLATE utf8mb4_bin NOT NULL,
  `user_type` smallint NOT NULL,
  `is_active` smallint NOT NULL DEFAULT '1',
  `initiative_id` int unsigned DEFAULT NULL,
  `citizen_id` int unsigned DEFAULT NULL,
  `created_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `username` (`username`),
  KEY `initiative_id` (`initiative_id`),
  KEY `citizen_id` (`citizen_id`),
  CONSTRAINT `users_ibfk_1` FOREIGN KEY (`initiative_id`) REFERENCES `initiatives` (`id`) ON DELETE SET NULL ON UPDATE RESTRICT,
  CONSTRAINT `users_ibfk_2` FOREIGN KEY (`citizen_id`) REFERENCES `citizens` (`id`) ON DELETE SET NULL ON UPDATE RESTRICT
) ENGINE=InnoDB AUTO_INCREMENT=6 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_bin;
